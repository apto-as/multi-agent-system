# src/services/tool_discovery_service.py
class ToolDiscoveryService:
    async def discover_tools(
        self,
        agent_id: str,
        namespace: str,  # âœ… Verified from DB (like V-VERIFY-3)
        category: Optional[str] = None
    ) -> List[MCPTool]:
        """
        Discover tools available to agent in their namespace

        Security (P0-1 pattern):
        - namespace verified from database (not user input)
        - tools filtered by namespace access_level
        """
        # V-VERIFY-3 pattern: Verify namespace from DB
        agent = await self.db.query(Agent).filter_by(id=agent_id).first()
        verified_namespace = agent.namespace  # âœ… Never trust user input

        # Query tools accessible in this namespace
        query = select(MCPTool).where(
            or_(
                MCPTool.access_level == "PUBLIC",
                MCPTool.access_level == "SYSTEM",
                and_(
                    MCPTool.access_level == "TEAM",
                    MCPTool.namespace == verified_namespace
                ),
                and_(
                    MCPTool.access_level == "PRIVATE",
                    MCPTool.agent_id == agent_id
                )
            )
        )

        if category:
            query = query.where(MCPTool.category == category)

        return await self.db.execute(query).scalars().all()
